WITH ts AS (
  SELECT teamID, yearID, SUM(salary) AS total_spend
  FROM salaries
  GROUP BY teamID, yearID
)
SELECT teamID, ROUND(AVG(total_spend)/1000000, 2) AS avg_spend_millions
FROM ts
GROUP BY teamID
ORDER BY avg_spend_millions DESC
LIMIT 5;
